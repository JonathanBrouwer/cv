	\ultracompactcventry
	{Quadrupel: Control Software for a Quadcopter Drone}
	{\begin{cvitems}
		\item Controlled using no-std Rust running on a Nordic nRF51 microcontroller.
		\item Achieves remarkable stability only using simple PID controllers.
		\item Developed for Embedded Systems Lab, a TU Delft course meant to teach students Rust.
	\end{cvitems}}
	
	\ultracompactcventry
	{Lody: A Dependently Typed Programming Language}
	{\begin{cvitems}
		\item A fully-featured dependently typed language, declaratively defined using Spoofax.
		\item Dependent types allow the user to write proofs about their code, guaranteeing correctness.
		\item Developed for my master thesis, which earned the cum laude distinction.
	\end{cvitems}}
